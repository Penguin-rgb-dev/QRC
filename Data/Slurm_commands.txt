#Submit a job via slurm script
sbatch submit.sh

# Check the status of all the nodes
sinfo

#check the status of all running jobs
squeue

# status of the user's job
squeue --me

# details like time taken by a job after it is complete
sacct -j <jobID> --format=JobID,JobName,MaxRSS,Elapsed,State

#Copy
cp original_file_name new_file_name

#How to check QOS limits
sacctmgr show qos long format=Name,MaxWall,MaxTRESpu

# To check the priority score and its breakdown for all the pending jobs
sprio

# Transfer multiple files to local machine
Method 1
scp 'divesh@172.27.36.235:/home/divesh/QRC/{filename1.ext,filename2.ext}' .

Method 2 (use wildcard *)
scp divesh@172.27.36.235:/home/divesh/QRC/convergence_plot_\*.npz .
or
scp divesh@172.27.36.235:/home/divesh/QRC/*.npz .

Method 3 (copy entire directory)
scp -r divesh@172.27.36.235:/home/divesh/QRC .

# Transfer large files
rsync -avzP divesh@172.27.36.235:/home/divesh/QRC/*.npz .

# Tranfer to the HPC
scp convergence_plot_1 divesh@172.27.36.235:/home/divesh/QRC
